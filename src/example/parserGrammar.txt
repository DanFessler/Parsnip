LITERAL: <string> | <number>

STRING: "([^"\\]|\\.)*"

NUMBER: \d+

IDENTIFIER: [a-zA-Z_][a-zA-Z0-9_]*

EXPRESSION: <ADD>

ADD: <MUL> ["+" | "-"] <ADD>

MUL: <VALUE> ["*" | "/" | "%"] <MUL>

VALUE: <LITERAL> | <IDENTIFIER> | <CALL> | <GROUP>

GROUP: "(" <EXPRESSION> ")"

CALL: <IDENTIFIER> "(" <ARGUMENTS> ")"

ARGUMENTS: <EXPRESSION> ["," <EXPRESSION>]*
